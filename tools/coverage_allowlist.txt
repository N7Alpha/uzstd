# Coverage exclusion allowlist for uzstd.h
#
# Each non-comment, non-blank line below is the exact stripped source text of a
# line in uzstd.h that carries one or more uncovered branch sides (or, in the one
# "line artifact" case below, an uncovered line) which are unreachable in
# practice. tools/coverage.sh treats misses on these lines as expected, so the
# gate can report 100% over *reachable* code. Any uncovered line or branch NOT
# listed here fails the gate (regression detection).
#
# These are defensive / dead-by-invariant branches. Per project policy the
# decoder's hardening checks against malformed input are kept in place (not
# deleted) even though specific sides cannot fire; they are documented here
# instead. If you edit one of these lines, update its text below.
#
# --- encoder: match finder / parse ---
# lazy-match insert cursor is maintained at exactly i+1 by the outer insert loop.
if (ins <= i+1) { UZSTD__INS(ins); ins++; }
#
# --- encoder: Huffman code-length Kraft repair ---
# inner loops always terminate via the index bound; the K>C / K<C side of the
# compound condition is re-tested by the enclosing while, never as the for exit.
for (i = 0; i < m && K > C; i++) {
for (i = m-1; i >= 0 && K < C; i--) {
#
# --- encoder: Huffman tree (weights) description ---
# weight FSE accuracy log is <= 5 for <= 256 symbols, so it is never clamped to 6.
if (log > 6) log = 6;
# write_ncount of <= 12 weight symbols always fits 127 bytes, so ncsz is never 0.
if (ncsz) {
# FSE-coded weights always fit the 127-byte budget for <= 256 symbols, so the
# direct (nibble) description is never preferred and bssz is never 0.
if (bssz && ncsz + bssz <= 127 && (nw > 128 || 1 + ncsz + bssz < direct)) {
# same: with FSE always chosen for > 128 symbols, the direct-path reject is dead.
if (nw > 128) return 0;
#
# --- encoder: literals section ---
# distinct >= 2 here (the distinct == 1 RLE case returned above), so huf_lengths
# never returns 0 and canonical codes always complete.
if (n >= 64 && (maxbits = uzstd__huf_lengths(cnt, &last_sym, len)) != 0
&& uzstd__huf_canonical(len, code, last_sym, maxbits)) {
# the cap >= rawh + n + 8 guard above guarantees the Huffman stream(s) always fit,
# so the 1-stream / 4-stream overflow (zero-size) returns cannot fire.
if (ssz) csize = tsz + ssz;
s2 = s1 ? uzstd__huf_stream(p, bend, L+q, q, code, len) : 0; p += s2;
s3 = s2 ? uzstd__huf_stream(p, bend, L+2*q, q, code, len) : 0; p += s3;
s4 = s3 ? uzstd__huf_stream(p, bend, L+3*q, n-3*q, code, len) : 0;
if (s4 && s1 <= 0xFFFF && s2 <= 0xFFFF && s3 <= 0xFFFF) {
#
# --- encoder: block driver ---
# the scratch buffer is sized so literal/sequence encoding into it never fails.
if (sz) {
#
# --- public compress: argument check ---
# the src_size >= 2 GiB reject needs a ~2 GiB input, outside the fuzz size budget.
if (dst_cap < UZSTD_COMPRESS_BOUND(src_size) || src_size >= 0x7FFFFFF0u) return (size_t)-1;
#
# --- decoder: defensive guards (kept for hardening; specific sides unreachable) ---
# the byte-position overflow side trips only with a 0xFFFF zero-run flag in a
# <4-byte table; the symbol-overflow side fires first in practice.
if (sym > max_sym || (bp>>3) > avail) return 0;
# FSE count widths are bounded by the running threshold, so remaining never < 1.
if (remaining < 1 || sym > max_sym) return 0;
# read_ncount only succeeds for a complete distribution, which fse_dtable always
# accepts, so the !fse_dtable side never fires.
if (!nc || !uzstd__fse_dtable(dt, norm, 15, log)) return 0;
if (!nc || !uzstd__fse_dtable(dt, norm, mxs[k], log)) return (size_t)-1;
# the RFC predefined distributions are valid, so fse_dtable never fails on them.
if (!uzstd__fse_dtable(dt, dnorm[k], dmxs[k], dlog[k])) return (size_t)-1;
# seq_prep returns at most the bytes available, so used > remaining never holds.
if (UZSTD_IS_ERROR(used) || used > (size_t)(iend - ip)) return 0;
# reached only after a fully valid sequence section leaves trailing bits; the
# unbounded fuzzer exercises it, the capped CI run does not deterministically.
if (b.err || b.pos != 0) return 0; /* bitstream must be fully consumed */

# --- llvm-cov line-mapping artifact (line coverage) ---
# The final fall-through return of uzstd__encode_sequences is reported as executed
# 0 times even though every ns>0 call returns through it (verified: the line above
# and the closing brace both show the full count). This is an llvm-cov counter-
# mapping quirk for a return that duplicates the function's earlier early-return
# text, not dead code. NOTE: this is the only `return (size_t)(d - dst);` that is
# ever uncovered; the identical returns in uzstd__frame_header / uzstd_compress are
# covered, so if one of those regresses this entry would mask it.
return (size_t)(d - dst);
